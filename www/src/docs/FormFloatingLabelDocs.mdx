# Floating labels

<p className="lead">
  Create beautifully simple form labels that float over your input fields.
</p>

<div className="row gy-5 pt-3">

  <Example title="Example">
    <Example.Intro>
      Wrap a `<Form.Control>` element in `<FloatingLabel>` to enable floating labels with
      Bootstrap’s textual form fields. A `placeholder` is required
      on each `<Form.Control>` as our method of CSS-only
      floating labels uses the `:placeholder-shown` pseudo-element.
    </Example.Intro>
    <Example.Preview>
      <FloatingLabel controlId="floatingInput" label="Email address" className="mb-3">
        <Form.Control type="email" placeholder="name@example.com" />
      </FloatingLabel>

      <FloatingLabel controlId="floatingPassword" label="Password">
        <Form.Control type="password" placeholder="Password" />
      </FloatingLabel>
    </Example.Preview>
    <Example.Code>
      ```js
      <FloatingLabel controlId="floatingInput" label="Email address" className="mb-3">
        <Form.Control type="email" placeholder="name@example.com" />
      </FloatingLabel>

      <FloatingLabel controlId="floatingPassword" label="Password">
        <Form.Control type="password" placeholder="Password" />
      </FloatingLabel>
      ```
    </Example.Code>
  </Example>

  <Example title="Textareas">
    <Example.Intro>
      By default, `<textarea>`s will be the same height as `<input>`s. To set a custom
      height on your `<textarea>`, do not use the `rows` attribute. Instead, set an
      explicit `height` (either inline or via custom CSS).
    </Example.Intro>
    <Example.Preview>
      <FloatingLabel controlId="floatingTextarea" label="Comments" className="mb-3">
        <Form.Control as="textarea" placeholder="Leave a comment here" />
      </FloatingLabel>

      <FloatingLabel controlId="floatingTextarea2" label="Comments">
        <Form.Control as="textarea" placeholder="Leave a comment here" style={{ height: '100px' }} />
      </FloatingLabel>
    </Example.Preview>
    <Example.Code>
      ```js
      <FloatingLabel controlId="floatingTextarea" label="Comments" className="mb-3">
        <Form.Control as="textarea" placeholder="Leave a comment here" />
      </FloatingLabel>

      <FloatingLabel controlId="floatingTextarea2" label="Comments">
        <Form.Control as="textarea" placeholder="Leave a comment here" style={{ height: '100px' }} />
      </FloatingLabel>
      ```
    </Example.Code>
  </Example>

  <Example title="Selects">
    <Example.Intro>
      Other than `<Form.Control>`, floating labels are only available on `<Form.Select>`s.
      They work in the same way, but unlike `<input>`s, they’ll always show the `<label>`
      in its floated state.
    </Example.Intro>
    <Example.Preview>
      <FloatingLabel controlId="floatingSelect" label="Works with selects">
        <Form.Select aria-label="Floating label select example">
          <option>Open this select menu</option>
          <option value="1">One</option>
          <option value="2">Two</option>
          <option value="3">Three</option>
        </Form.Select>
      </FloatingLabel>
    </Example.Preview>
    <Example.Code>
      ```js
      <FloatingLabel controlId="floatingSelect" label="Works with selects">
        <Form.Select aria-label="Floating label select example">
          <option>Open this select menu</option>
          <option value="1">One</option>
          <option value="2">Two</option>
          <option value="3">Three</option>
        </Form.Select>
      </FloatingLabel>
      ```
    </Example.Code>
  </Example>

  <Example title="Layout">
    <Example.Intro>
      When working with the Bootstrap grid system, be sure to place form
      elements within column classes.
    </Example.Intro>
    <Example.Preview>
      <Row className="g-2">
        <Col md>
          <FloatingLabel controlId="floatingInputGrid" label="Email address">
            <Form.Control type="email" placeholder="name@example.com" />
          </FloatingLabel>
        </Col>
        <Col md>
          <FloatingLabel controlId="floatingSelectGrid" label="Works with selects">
            <Form.Select aria-label="Floating label select example">
              <option>Open this select menu</option>
              <option value="1">One</option>
              <option value="2">Two</option>
              <option value="3">Three</option>
            </Form.Select>
          </FloatingLabel>
        </Col>
      </Row>
    </Example.Preview>
    <Example.Code>
      ```js
      <Row className="g-2">
        <Col md>
          <FloatingLabel controlId="floatingInputGrid" label="Email address">
            <Form.Control type="email" placeholder="name@example.com" />
          </FloatingLabel>
        </Col>
        <Col md>
          <FloatingLabel controlId="floatingSelectGrid" label="Works with selects">
            <Form.Select aria-label="Floating label select example">
              <option>Open this select menu</option>
              <option value="1">One</option>
              <option value="2">Two</option>
              <option value="3">Three</option>
            </Form.Select>
          </FloatingLabel>
        </Col>
      </Row>
      ```
    </Example.Code>
  </Example>

  <Example title="Customizing rendering">
    <Example.Intro>
      If you need greater control over the rendering, use the `<FormFloating>` component
      to wrap your input and label. Also note that the `<Form.Control>` must come first
      so we can utilize a sibling selector (e.g., ~).
    </Example.Intro>
    <Example.Preview>
      <FormFloating className="mb-3">
        <Form.Control id="floatingInputCustom" type="email"placeholder="name@example.com" />
        <label htmlFor="floatingInputCustom">Email address</label>
      </FormFloating>

      <FormFloating>
        <Form.Control id="floatingPasswordCustom" type="password" placeholder="Password" />
        <label htmlFor="floatingPasswordCustom">Password</label>
      </FormFloating>
    </Example.Preview>
    <Example.Code>
      ```js
      <FormFloating className="mb-3">
        <Form.Control id="floatingInputCustom" type="email"placeholder="name@example.com" />
        <label htmlFor="floatingInputCustom">Email address</label>
      </FormFloating>

      <FormFloating>
        <Form.Control id="floatingPasswordCustom" type="password" placeholder="Password" />
        <label htmlFor="floatingPasswordCustom">Password</label>
      </FormFloating>
      ```
    </Example.Code>
  </Example>
